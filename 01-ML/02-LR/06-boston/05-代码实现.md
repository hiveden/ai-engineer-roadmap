```python
def linear_model1():
    # 1 获取数据
    data = load_boston()  # 无法下载，需手动加载

    # 2 数据集划分
    x_train, x_test, y_train, y_test = train_test_split(
        data.data, data.target, random_state=22)

    # 3 特征工程-标准化
    transfer = StandardScaler()
    x_train = transfer.fit_transform(x_train)
    x_test = transfer.transform(x_test)

    # 4 机器学习-线性回归（正规方程）
    estimator = LinearRegression()
    estimator.fit(x_train, y_train)

    # 5 模型评估
    y_predict = estimator.predict(x_test)
    print("预测值为:\n", y_predict)
    print("模型的权重系数为:\n", estimator.coef_)
    print("模型的偏置为:\n", estimator.intercept_)

    # 5.2 评价 均方误差
    error = mean_squared_error(y_test, y_predict)
    print("误差为:\n", error)
    # 思考：为啥正规方程法的预测误差小？
```

## ━━━━━━━━ 讲解 ━━━━━━━━

### 完整代码（正规方程 + SGD 对比）

```python
"""
线性回归 · 加州房价预测
Boston 数据集已废弃，本脚本用 California 演示同样的流程。
"""
from sklearn.datasets import fetch_california_housing
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression, SGDRegressor
from sklearn.metrics import mean_squared_error, mean_absolute_error, r2_score
import numpy as np


def run():
    # 1 加载数据
    data = fetch_california_housing()
    X, y = data.data, data.target          # X: (20640, 8), y: 房价 / 10 万美元
    print(f"特征: {data.feature_names}")
    print(f"X shape: {X.shape}, y shape: {y.shape}")

    # 2 拆分（回归任务不用 stratify）
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=0.2, random_state=22,
    )

    # 3 标准化（先 fit_transform 训练集，再 transform 测试集）
    scaler = StandardScaler()
    X_train_s = scaler.fit_transform(X_train)
    X_test_s = scaler.transform(X_test)

    # 4-A 训练 · 正规方程
    model_ne = LinearRegression()
    model_ne.fit(X_train_s, y_train)

    # 4-B 训练 · 随机梯度下降
    model_sgd = SGDRegressor(
        loss="squared_error",
        learning_rate="invscaling",
        eta0=0.01,
        max_iter=1000,
        random_state=22,
    )
    model_sgd.fit(X_train_s, y_train)

    # 5 评估
    for name, model in [("正规方程", model_ne), ("SGD", model_sgd)]:
        y_pred = model.predict(X_test_s)
        mse = mean_squared_error(y_test, y_pred)
        mae = mean_absolute_error(y_test, y_pred)
        rmse = np.sqrt(mse)
        r2 = r2_score(y_test, y_pred)
        print(f"\n[{name}]")
        print(f"  MSE  = {mse:.3f}")
        print(f"  MAE  = {mae:.3f}")
        print(f"  RMSE = {rmse:.3f}")
        print(f"  R²   = {r2:.3f}")
        print(f"  权重: {model.coef_}")
        print(f"  截距: {model.intercept_}")

    # 6 预测新样本（注意：必须用同一个 scaler）
    new_x = np.array([[8.3, 41, 6.9, 1.0, 322, 2.5, 37.88, -122.23]])
    new_x_s = scaler.transform(new_x)
    print(f"\n预测房价: {model_ne.predict(new_x_s)[0] * 100000:.0f} 美元")


if __name__ == "__main__":
    run()
```

```
  RMSE = 0.745
  R²   = 0.596
  权重: [ 0.83 0.12 -0.27 0.31 -0.0 -0.04 -0.90 -0.87]

[SGD]
```

### 输出怎么读

**两种解法结果几乎一致**——验证了「LR 是凸优化，正规方程和 SGD 都收敛到同一最优解」。差异在 0.002 量级，是 SGD 随机性导致的。

**权重符号告诉你影响方向**（标准化后的权重直接可比）：

| 特征 | 权重 | 解读 |
|---|---|---|
| MedInc（收入中位数） | +0.83 | 收入越高房价越高 ✓ |
| Latitude（纬度） | -0.90 | 纬度越高（越往北）房价越低 ✓ 加州中心在南部 |
| Longitude（经度） | -0.87 | 经度越大（越往东）房价越低 ✓ 沿海更贵 |
| AveRooms | -0.27 | 反直觉？因为已被 MedInc 解释了，残留是「房间多但收入低」的农村效应 |

> Tech Lead 视角：**永远要在标准化后的权重上做特征重要性分析**，原始权重受量纲影响，比不出大小。

### 为什么 LR 误差小？（PPT 思考题答案）

PPT slide 95 留了个思考：「为啥正规方程法的预测误差小？」——准确说**两种方法误差几乎一样**。原 PPT 在 Boston 小数据上跑出微小差异是 SGD 没收敛完全，加大 `max_iter` 或换 `learning_rate="adaptive"` 就一致了。

**真正的差异**不在精度，而在：

| 维度 | 正规方程 | SGD |
|---|---|---|
| 时间复杂度 | O(n·p² + p³) | O(n·p·iter) |
| 内存 | 必须把 X 全装内存 | 可流式喂 batch（partial_fit） |
| 大数据 | 100 万样本就吃力 | 上亿样本依然可用 |

### 部署到生产环境

LR 部署比深度模型简单一万倍——**只要保存 `coef_` 和 `intercept_` 就够**。

#### 方案 A · pickle 整个 sklearn 模型

```python
import joblib

joblib.dump({"scaler": scaler, "model": model_ne}, "lr.pkl")

# 推理端（必须装 sklearn）
bundle = joblib.load("lr.pkl")
y_hat = bundle["model"].predict(bundle["scaler"].transform(new_x))
```

简单但耦合 sklearn 版本——升级 sklearn 经常 unpickle 不了。

#### 方案 B · 导出权重，跨语言推理

```python
import json

bundle = {
    "feature_names": list(data.feature_names),
    "scaler_mean": scaler.mean_.tolist(),
    "scaler_scale": scaler.scale_.tolist(),
    "coef": model_ne.coef_.tolist(),
    "intercept": float(model_ne.intercept_),
}
with open("lr.json", "w") as f:
    json.dump(bundle, f, indent=2)
```

然后在 Go / Java / JS 里**手写一次预测函数**就能直接上线，无需任何 ML 库：

```go
// Go 推理（零依赖）
func predict(x []float64, b Bundle) float64 {
    sum := b.Intercept
    for i, xi := range x {
        xs := (xi - b.ScalerMean[i]) / b.ScalerScale[i]
        sum += b.Coef[i] * xs
    }
    return sum
}
```

```javascript
// JS 推理（浏览器端 / Node 都能跑）
const predict = (x, b) => {
  const xs = x.map((xi, i) => (xi - b.scaler_mean[i]) / b.scaler_scale[i]);
  return xs.reduce((s, v, i) => s + b.coef[i] * v, b.intercept);
};
```

#### 方案 C · ONNX（标准化的工程路径）

```python
from skl2onnx import convert_sklearn
from skl2onnx.common.data_types import FloatTensorType

onx = convert_sklearn(
    model_ne,
    initial_types=[("input", FloatTensorType([None, 8]))],
)
with open("lr.onnx", "wb") as f:
    f.write(onx.SerializeToString())
```

ONNX 可以在 Python / Go / Java / Node / C++ / Rust / 浏览器（onnxruntime-web）所有主流 runtime 上加载。**对 LR 来说是杀鸡用牛刀**（方案 B 已经够），但学一遍流程，因为后面跑 NN 时候必走 ONNX。
